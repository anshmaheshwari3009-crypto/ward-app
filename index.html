    <script>

        const snapBtn = document.getElementById('snapBtn');

        const cameraInput = document.getElementById('cameraInput');

        const previewContainer = document.getElementById('previewContainer');

        const imagePreview = document.getElementById('imagePreview');

        const fileSizeLabel = document.getElementById('fileSizeLabel');

        const clearBtn = document.getElementById('clearBtn');

        const form = document.getElementById('uploadForm');

        const submitBtn = document.getElementById('submitBtn');

        const statusAlert = document.getElementById('statusAlert');

        const historyList = document.getElementById('historyList');


        let selectedFile = null;

        let fileDataUrl = null; 


        // Load history from browser memory

        let uploads = JSON.parse(localStorage.getItem('hospital_uploads')) || [];

        renderHistory();


        snapBtn.addEventListener('click', () => cameraInput.click());


        cameraInput.addEventListener('change', (e) => {

            const file = e.target.files[0];

            if (file) {

                selectedFile = file;

                

                // Convert to Base64 for local storage simulation

                const reader = new FileReader();

                reader.onload = (e) => {

                    fileDataUrl = e.target.result;

                    imagePreview.src = fileDataUrl;

                };

                reader.readAsDataURL(file);


                fileSizeLabel.textContent = (file.size / (1024 * 1024)).toFixed(2) + ' MB';

                previewContainer.classList.remove('hidden');

                snapBtn.classList.add('hidden');

            }

        });


        clearBtn.addEventListener('click', () => {

            selectedFile = null;

            fileDataUrl = null;

            cameraInput.value = '';

            previewContainer.classList.add('hidden');

            snapBtn.classList.remove('hidden');

        });


        form.addEventListener('submit', (e) => {

            e.preventDefault();

            if (!selectedFile) return alert('Please snap a picture of the report first.');


            // UI loading state

            submitBtn.disabled = true;

            submitBtn.innerHTML = '<span>Processing & Routing...</span>';

            statusAlert.classList.add('hidden');


            // Simulate network delay of 1.5 seconds

            setTimeout(() => {

                const uhid = document.getElementById('uhid').value.trim();

                const patientName = document.getElementById('patientName').value.trim();

                const doctorContact = document.getElementById('doctorContact').value.trim();

                const safeUhid = uhid.replace(/[^a-zA-Z0-9-_]/g, '_');

                

                const newUpload = {

                    id: Date.now(),

                    uhid: safeUhid,

                    name: patientName || 'Unknown',

                    doctor: doctorContact,

                    image: fileDataUrl,

                    time: new Date().toLocaleTimeString()

                };


                uploads.unshift(newUpload);

                localStorage.setItem('hospital_uploads', JSON.stringify(uploads));


                showStatus(`✅ Routed to folder <b>/patients/${safeUhid}/</b><br>Notification sent to ${doctorContact}`, 'bg-green-100 text-green-800 border border-green-200');

                

                form.reset();

                clearBtn.click();

                renderHistory();

                

                submitBtn.disabled = false;

                submitBtn.innerHTML = '<span>Dispatch to Senior</span>';

            }, 1500);

        });


        function showStatus(msg, classes) {

            statusAlert.className = 'mt-4 p-4 rounded-xl text-sm font-medium text-center shadow-sm ' + classes;

            statusAlert.innerHTML = msg;

            statusAlert.classList.remove('hidden');

        }


        function renderHistory() {

            if (uploads.length === 0) return;

            

            historyList.innerHTML = uploads.map(upload => `

                <div class="bg-white p-3 rounded-xl border border-slate-200 shadow-sm flex space-x-3 items-center">

                    <img src="${upload.image}" class="w-16 h-16 object-cover rounded-lg border border-slate-200">

                    <div class="flex-1 overflow-hidden">

                        <div class="flex justify-between items-start">

                            <h3 class="font-bold text-slate-800 truncate text-sm">📁 /${upload.uhid}</h3>

                            <span class="text-[10px] font-semibold text-slate-400 bg-slate-100 px-2 py-1 rounded">${upload.time}</span>

                        </div>

                        <p class="text-xs text-slate-500 mt-0.5 truncate">Patient: ${upload.name}</p>

                        <p class="text-xs text-blue-600 mt-0.5 truncate font-medium">Alerted: ${upload.doctor}</p>

                    </div>

                </div>

            `).join('');

        }
